    const sliderWrap = document.querySelector(".slider__wrap");         // 이미지 영역을 감싸는 전체 영역
    const sliderImg = document.querySelector(".slider__img");           // 이미지가 보이는 영역
    const sliderInner = document.querySelector(".slider__inner");       // 이미지를 움직이게 하는 영역
    const slider = document.querySelectorAll(".slider");                // 5개의 이미지가 저장

    let currentIndex = 0;                                               // 첫번째 이미지, 현재 보여지는 이미지
    let sliderCount = slider.length;                                    // 저장된 이미지 개수
    let sliderClone = sliderInner.firstElementChild.cloneNode(true);    // 이미지가 보이는 영역의 첫번째 자식 요소(sliderinner의 첫번째 요소)가 있으면 복사해서 반환
    sliderInner.appendChild(sliderClone);                               // 첫번째 이미지를 마지막에 넣어준다
    let sliderWidth = sliderImg.offsetWidth;                            // 이미지의 가로 길이

    function sliderEffect() {
        currentIndex++;                                                 // 현재 인덱스 값을 1 증가(다음 이미지의 인덱스)
        sliderInner.style.transition = "all 0.3s";                      // 이미지 움직임의 지연시간을 0.3초로 설정
        sliderInner.style.transform = "translateX(-" + sliderWidth * currentIndex + "px)";  // (-가로 방향으로 길이 * 현재 인덱스)만큼 이동

        // sliderInner.style.transform = "translateX(-800)";
        // sliderInner.style.transform = "translateX(-1600)";
        // sliderInner.style.transform = "translateX(-2400)";
        // sliderInner.style.transform = "translateX(-3200)";
        // sliderInner.style.transform = "translateX(-3800)";

        //마지막 이미지로 갔을때
        if(currentIndex == sliderCount) {
            setTimeout(() => {                                         // setTimeOut() : 지정한 시간이 지나면(300) 대입한 콜백 함수를 실행
                // 초기화
                sliderInner.style.transform = "translateX(0)";         // 가로 위치를 0으로 초기화(첫번째 이미지)
                sliderInner.style.transition = "0s";                   // 이미지 이동 지연시간을 0초로 초기화(자연스러운 움직임)
            }, 300);                                                   // 0.3초 뒤에 실행
            currentIndex = 0;                                          // 현재 이미지를 첫번째 이미지로 변경
        }

    }

    setInterval(sliderEffect, 2000);
