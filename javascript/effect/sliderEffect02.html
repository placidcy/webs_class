    const sliderWrap = document.querySelector(".slider__wrap");         // 이미지 영역을 감싸는 전체 영역
    const sliderImg = document.querySelector(".slider__img");           // 이미지가 보이는 영역
    const sliderInner = document.querySelector(".slider__inner");       // 이미지를 움직이게 하는 영역
    const slider = document.querySelectorAll(".slider");                // 5개의 이미지가 저장

    let currentIndex = 0;               // 첫번째 이미지, 현재 보여지는 이미지
    let sliderCount = slider.length;     // 이미지 개수

    // setInterval() : 특정 시간마다 자동으로 실행하는 함수(3000 -> 3초마다 실행)
    setInterval(() => {                                                
        (currentIndex < sliderCount - 1) ? currentIndex++ : currentIndex = 0;    // sliderCount-1 : 0, 4번지까지 이미지가 저장되어 있기 때문에 -1을 뺀다

        gsap.to(sliderInner, {duration: 0.4, left: (-800 * currentIndex), ease: "expo.out"});   // GSAP 애니메이션()
                                                                                                //-800 * currentindex : 시작 이미지 0부터 4까지 0px에서 -3200px까지 sliderInner(보여지는 영역)의 위치를 왼쪽으로 -800px 이동한다

    }, 3000);
