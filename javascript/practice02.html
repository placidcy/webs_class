02. 구구단 출력하기

1.1 1단부터 9단까지 출력하기

{
    for (let i = 2; i <= 9; i++) {
        document.write("<br>");

        for (let j = 1; j <= 9; j++) {
            let value = i * j;

            document.write(i + " x " + j + " = " + value, "<br>");
        }
    }
}

1.2. while문을 이용해서 구구단 출력

{
    let num1 = 2;
    let num2 = 1;
    let value = 0;

    while(num1 <= 9 && num2 <= 9){
        value = num1 * num2;

        document.write(num1 + " x " + num2 + " = " + value);
        num2++;

        if(num2 > 9){
            num1++;
            num2 = 1;
        }
        document.write("<br>");
    }
}

1.3. 사용자가 원하는 단을 받아 출력

사용자가 원하는 단을 promit() 함수를 사용합니다클릭

{
    let num, result;

    document.querySelector(".test .btn03").addEventListener("click", () => {
        num = prompt("출력할 구구단 숫자를 입력하세요.", "2");

        for(let i = 1; i <= 9; i++){
            result = num + " * " + i +  " = " + (num * i); + "<br>>";
        }
    });
}
}
